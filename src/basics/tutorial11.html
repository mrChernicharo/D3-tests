Some Array functions provided by d3

const data = [10, 20, 30, 40, 50, 60];


d3.min(data)
// 10
d3.max(data)
// 60
data.sort(d3.descending)
// (6) [60, 50, 40, 30, 20, 10]
d3.extent(data)
// (2) [10, 60]
d3.sum(data)
// 210
d3.mean(data)
// 35
d3.shuffle(data)
// (6) [20, 10, 50, 40, 60, 30]
d3.shuffle(data)
// (6) [30, 20, 40, 60, 50, 10]
data.sort(d3.ascending)
// (6) [10, 20, 30, 40, 50, 60]
